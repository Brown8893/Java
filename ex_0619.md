package myJava.ch06;
import java.lang.*;

public class Main
{
	public static void main(String arg[])
	{
		System.out.print("費式數列如下喔:");
		for(int i=0;i<=25;i++)
		{
			if(i%8==0)
				System.out.println();
				System.out.print(Fib(i) + "\t");
		}
		System.out.println("......");
	}
	
	public static int Fib(int n)
	{
		if((n==1) || (n==0))
			return n;
		else
			return Fib(n-1) + Fib(n-2);
	}
}

/**
 * @(#)Main.java

 * Main application

 * @author

/*  檔名:ch6_19.java        功能:遞迴函式求費氏數列  */
package myJava.ch06;
import java.lang.*;

public class Main
{
	public static void main(String arg[])
	{
		System.out.print("費式數列如下喔:");
		for(int i=0;i<=44;i++)
		{
			/*if(i%8==0)
				System.out.println();*/
				System.out.print(Fib(i) + "\t");
				System.out.println(Fib2(i) + "\t");
		}
		System.out.println("......");
	}
	
	public static int Fib2(int n)
	{
		int i,f1,f2,f3;
		if((n==1) || (n==0))
		{
			return n;
		}
		else
		{
			f1=0;f2=1;f3=0;
			for(i=2;i<=n;i++)
			{
				f3=f1+f2;
				f1=f2;
				f2=f3;
			}
			return f3;
		}
	}
	
	
	public static int Fib(int n)
	{
		if((n==1) || (n==0))
			return n;
		else
			return Fib(n-1) + Fib(n-2);
	}
}
